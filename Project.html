DCMthetaLambda = sp.lambdify(theta, DCMtheta, 'numpy')

class Basketball(ThreeDScene):
    def construct(self):
        self.set_camera_orientation(phi=75 * DEGREES, theta=-45 * DEGREES)

        axes = ThreeDAxes()
        bodyAxes = ThreeDAxes(x_length=2, y_length=2, z_length=2, stroke_color=BLUE)
        # set length of body axes to .1 and make them blue
        sphere = Sphere(radius=1, fill_opacity=.7)
        sphere.shift(1*OUT)

        self.add(axes, bodyAxes, sphere)

        # orient using the direction cosine matrix
        def update_sphere(mob, dt):
            bodya = (DCMphiLambda(interp_phi(dt)) @ np.array([[0], [0], [1]])).T[0]
            bodyb = (DCMthetaLambda(interp_theta(dt)) @ DCMphiLambda(interp_phi(dt)) @ np.array([[1], [0], [0]])).T[0]
            bodyc = (DCMpsiLambda(interp_psi(dt)) @ DCMthetaLambda(interp_theta(dt)) @ DCMphiLambda(interp_phi(dt)) @ np.array([[0], [0], [1]])).T[0]
            mob.rotate_about_origin(interp_psi(dt), bodya).rotate_about_origin(interp_phi(dt), bodyb).rotate_about_origin(interp_theta(dt), bodyc)

        def update_axes(mob, dt):
            bodya = (DCMphiLambda(interp_phi(dt)) @ np.array([[0], [0], [1]])).T[0]
            bodyb = (DCMthetaLambda(interp_theta(dt)) @ DCMphiLambda(interp_phi(dt)) @ np.array([[1], [0], [0]])).T[0]
            bodyc = (DCMpsiLambda(interp_psi(dt)) @ DCMthetaLambda(interp_theta(dt)) @ DCMphiLambda(interp_phi(dt)) @ np.array([[0], [0], [1]])).T[0]
            mob.rotate_about_origin(interp_psi(dt), bodya).rotate_about_origin(interp_phi(dt), bodyb).rotate_about_origin(interp_theta(dt), bodyc)

        sphere.add_updater(update_sphere)
        bodyAxes.add_updater(update_axes)

        self.play(FadeIn(sphere), run_time=.1)
        self.wait(5)
